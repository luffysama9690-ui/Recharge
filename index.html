const tg = window.Telegram.WebApp; tg.expand(); let selectedProduct = ""; let selectedPrice = 0; function selectItem(product, price, element) { document.querySelectorAll('.card').forEach(c => c.classList.remove('selected')); element.classList.add('selected'); selectedProduct = product; selectedPrice = price; document.getElementById('displayPrice').innerText = price + " MMK"; } function placeOrder() { const gId = document.getElementById('gameId').value; const zId = document.getElementById('zoneId').value; if(!gId || !zId) { tg.showAlert("Game ID နှင့် Zone ID ကို ဖြည့်ပေးပါခင်ဗျာ။"); return; } if(!selectedProduct) { tg.showAlert("ဝယ်ယူမည့် စိန်ပမာဏကို ရွေးချယ်ပေးပါဦး။"); return; } // ၁။ ဒေတာကို စုစည်းမယ် const orderData = { gameId: gId, zoneId: zId, product: selectedProduct, price: selectedPrice + " MMK" }; // ၂။ Telegram ရဲ့ တရားဝင်စနစ်အတိုင်း Bot ထဲကို ဒေတာ လှမ်းထည့်ပြီး ပိတ်ခွင့်ပြုလိုက်ပါ tg.sendData(JSON.stringify(orderData)); }
